    { name: 'The Host',         description: 'Crowd-pleasers, batched builds, easy to serve at scale.' },

  ];

  grid.style.cssText = 'display:grid;grid-template-columns:repeat(auto-fill,minmax(240px,1fr));gap:8px;';
  ARCHETYPES.forEach(a => {

  });
